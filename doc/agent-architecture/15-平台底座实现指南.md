## 三、`platform/manifestValidator.ts` — 校验器

```typescript
import { Effect, Schema } from "effect"
import {
  AgentManifest,
  CapabilityManifest,
  SkillManifest,
  type AnyManifest,
} from "./manifestSchema"
import { TOOL_REGISTRY } from "./toolRegistry"

export type ErrorCode =
  | "INVALID_API_VERSION"
  | "INVALID_KIND"
  | "INVALID_NAME"
  | "DUPLICATE_NAME"
  | "MISSING_REQUIRED_FIELD"
  | "UNKNOWN_TOOL"
  | "TOOL_NOT_PERMITTED_FOR_SOURCE"
  | "UNKNOWN_SKILL"
  | "CIRCULAR_INHERIT"
  | "PROMPT_TOO_SHORT"
  | "INVALID_PERMISSION_ACTION"
  | "AGENT_NOT_FOUND"
  | "UNKNOWN_CONTEXT_KEY"

export class ValidationError extends Error {
  constructor(public code: ErrorCode, public detail: unknown) {
    super(`[${code}] ${typeof detail === "string" ? detail : JSON.stringify(detail)}`)
  }
}

export function validate(manifest: AnyManifest): Effect.Effect<void, ValidationError> {
  return Effect.gen(function* () {
    // 1. Schema 静态校验
    switch (manifest.kind) {
      case "Agent":
        yield* Schema.decodeUnknown(AgentManifest)(manifest).pipe(
          Effect.mapError(
            (e) => new ValidationError("MISSING_REQUIRED_FIELD", e.message),
          ),
        )
        break
      case "Skill":
        yield* Schema.decodeUnknown(SkillManifest)(manifest).pipe(
          Effect.mapError(
            (e) => new ValidationError("MISSING_REQUIRED_FIELD", e.message),
          ),
        )
        break
      case "Capability":
        yield* Schema.decodeUnknown(CapabilityManifest)(manifest).pipe(
          Effect.mapError(
            (e) => new ValidationError("MISSING_REQUIRED_FIELD", e.message),
          ),
        )
        break
      default:
        return yield* Effect.fail(new ValidationError("INVALID_KIND", manifest))
    }

    // 2. Agent 业务规则校验
    if (manifest.kind === "Agent") {
      const m = manifest as AgentManifest
      // 工具白名单 × source 校验
      for (const tool of m.spec.tools.allowed) {
        const meta = TOOL_REGISTRY[tool]
        if (!meta) {
          return yield* Effect.fail(new ValidationError("UNKNOWN_TOOL", tool))
        }
        if (!meta.defaultAllowedSources.includes(m.metadata.source)) {
          return yield* Effect.fail(
            new ValidationError("TOOL_NOT_PERMITTED_FOR_SOURCE", {
              tool,
              source: m.metadata.source,
              allowed: meta.defaultAllowedSources,
            }),
          )
        }
      }
    }
  })
}
```

export class ToolRegistry {
  private tools = new Map<string, ToolMeta>()

  registerAll(map: Record<string, ToolMeta> = TOOL_REGISTRY) {
    for (const [k, v] of Object.entries(map)) this.tools.set(k, v)
  }

  get(name: string): ToolMeta | undefined {
    return this.tools.get(name)
  }

  list(filter?: { category?: ToolCategory; source?: ToolSource }): ToolMeta[] {
    return Array.from(this.tools.values()).filter((t) => {
      if (filter?.category && t.category !== filter.category) return false
      if (filter?.source && !t.defaultAllowedSources.includes(filter.source)) return false
      return true
    })
  }

  /** Loader 装载时强校验：source 是否允许使用某工具 */
  assertAllowed(toolName: string, source: ToolSource) {
    const meta = this.tools.get(toolName)
    if (!meta) throw new Error(`UNKNOWN_TOOL: ${toolName}`)
    if (!meta.defaultAllowedSources.includes(source)) {
      throw new Error(
        `TOOL_NOT_PERMITTED_FOR_SOURCE: tool=${toolName} source=${source} allowed=${meta.defaultAllowedSources.join(",")}`,
      )
    }
  }
}

/** 默认 source 对应的兜底权限规则 */
function defaultRules(source: AgentManifest["metadata"]["source"]) {
  if (source === "builtin") {
    return [{ permission: "*", pattern: "*", action: "allow" as const }]
  }
  // user / marketplace 只能 read 和 write 到 workspace
  return [
    { permission: "read",  pattern: "*",            action: "allow" as const },
    { permission: "write", pattern: "workspace/*",  action: "allow" as const },
    { permission: "write", pattern: "*",            action: "ask"   as const },
    { permission: "*",     pattern: "*",            action: "deny"  as const },
  ]
}

function manifestToAgentInfo(m: AgentManifest): Agent.Info {
  const fullPrompt = m.spec.contextTemplate
    ? `${m.spec.basePrompt}\n\n${m.spec.contextTemplate}`
    : m.spec.basePrompt

  return {
    name: m.metadata.name,
    description: m.metadata.description,
    mode: m.spec.mode,
    prompt: fullPrompt,
    permission: m.spec.permission ?? defaultRules(m.metadata.source),
    tools: m.spec.tools.allowed,
    temperature: m.spec.model?.temperature,
    model: m.spec.model
      ? { providerID: m.spec.model.provider, modelID: m.spec.model.modelID }
      : undefined,
  }
}

export class AgentLoader {
  load(manifest: AgentManifest): Effect.Effect<void, Error, Agent.Service | Permission.Service> {
    return Effect.gen(function* () {
      yield* validate(manifest)
      // 工具白名单逐个校验（source 维度）
      for (const t of manifest.spec.tools.allowed) {
        toolRegistry.assertAllowed(t, manifest.metadata.source)
      }
      const agentSvc = yield* Agent.Service
      const permSvc = yield* Permission.Service

      // 注册 Agent
      yield* agentSvc.register(manifestToAgentInfo(manifest))

      // 把 manifest 中的 permission rule merge 到全局 approved 列表
      if (manifest.spec.permission && manifest.spec.permission.length > 0) {
        yield* permSvc.setApproved([...manifest.spec.permission])
      }
    })
  }

  unload(name: string): Effect.Effect<void, Error, Agent.Service> {
    return Effect.gen(function* () {
      const agentSvc = yield* Agent.Service
      yield* agentSvc.unregister(name)
    })
  }

  reload(manifest: AgentManifest) {
    return Effect.gen(this, function* () {
      yield* this.unload(manifest.metadata.name)
      yield* this.load(manifest)
    })
  }
}

export const agentLoader = new AgentLoader()

## 七、`platform/capabilityRegistry.ts` — 能力注册中心

```typescript
import { Effect } from "effect"
import type { CapabilityManifest } from "./manifestSchema"
import { validate } from "./manifestValidator"

export interface CapabilityRecord {
  manifest: CapabilityManifest
  enabled: boolean
  installedAt: number
}

type Listener = (record: CapabilityRecord, action: "install" | "uninstall" | "enable" | "disable") => void

export class CapabilityRegistry {
  private records = new Map<string, CapabilityRecord>()
  private listeners = new Set<Listener>()

  subscribe(fn: Listener) {
    this.listeners.add(fn)
    return () => this.listeners.delete(fn)
  }

  private emit(record: CapabilityRecord, action: Parameters<Listener>[1]) {
    for (const fn of this.listeners) fn(record, action)
  }

  install(manifest: CapabilityManifest): Effect.Effect<void, Error> {
    const self = this
    return Effect.gen(function* () {
      yield* validate(manifest)
      if (self.records.has(manifest.metadata.name)) {
        throw new Error(`DUPLICATE_NAME: ${manifest.metadata.name}`)
      }
      const rec: CapabilityRecord = {
        manifest,
        enabled: true,
        installedAt: Date.now(),
      }
      self.records.set(manifest.metadata.name, rec)
      self.emit(rec, "install")
    })
  }

  uninstall(name: string): Effect.Effect<void, Error> {
    const self = this
    return Effect.gen(function* () {
      const rec = self.records.get(name)
      if (!rec) throw new Error(`Capability "${name}" not found`)
      if (rec.manifest.metadata.source === "builtin") {
        throw new Error(`Cannot uninstall builtin capability: ${name}; use disable instead`)
      }
      self.records.delete(name)
      self.emit(rec, "uninstall")
    })
  }

  enable(name: string) {
    const rec = this.records.get(name)
    if (!rec) return
    rec.enabled = true
    this.emit(rec, "enable")
  }

  disable(name: string) {
    const rec = this.records.get(name)
    if (!rec) return
    rec.enabled = false
    this.emit(rec, "disable")
  }

  get(name: string) {
    return this.records.get(name)
  }

  list(filter?: { source?: string; enabled?: boolean }): CapabilityRecord[] {
    return Array.from(this.records.values())
      .filter((r) => {
        if (filter?.source && r.manifest.metadata.source !== filter.source) return false
        if (filter?.enabled !== undefined && r.enabled !== filter.enabled) return false
        return true
      })
      .sort(
        (a, b) =>
          (a.manifest.spec.order ?? 999) - (b.manifest.spec.order ?? 999),
      )
  }
}

export const capabilityRegistry = new CapabilityRegistry()
```

## 九、`platform/bootstrap.ts` — 启动入口

```typescript
import { Effect, Layer } from "effect"
import { Agent } from "../agent-runtime/agent/agent"
import { Skill } from "../agent-runtime/skill/index"
import { Permission } from "../agent-runtime/permission/index"
import { agentLoader } from "./agentLoader"
import { skillLoader } from "./skillLoader"
import { capabilityRegistry } from "./capabilityRegistry"
import {
  BUILTIN_AGENT_MANIFESTS,
  BUILTIN_SKILL_MANIFESTS,
  BUILTIN_CAPABILITY_MANIFESTS,
} from "./builtinSeed"
import { invoke } from "@tauri-apps/api/core"

/**
 * 平台启动流程（vite/tauri 主入口启动 React 之前调用）
 * 1. 注册工具元数据（toolRegistry 已在文件顶层自动注册，这里仅校验）
 * 2. 加载内置 Skill / Agent / Capability
 * 3. 从 SQLite + 文件系统加载用户 Skill / Agent / Capability
 * 4. 暴露给前端 window.__platform.*
 */
export async function bootstrapPlatform() {
  const program = Effect.gen(function* () {
    // 内置 Skill 先于 Agent 加载
    for (const m of BUILTIN_SKILL_MANIFESTS) {
      yield* skillLoader.load(m)
    }
    for (const m of BUILTIN_AGENT_MANIFESTS) {
      yield* agentLoader.load(m)
    }
    for (const m of BUILTIN_CAPABILITY_MANIFESTS) {
      yield* capabilityRegistry.install(m)
    }
  })

  const layer = Layer.mergeAll(Agent.layer, Skill.layer, Permission.layer)
  await Effect.runPromise(program.pipe(Effect.provide(layer)))

  // 用户能力（从 SQLite + 文件系统扫描）
  const userManifests = await invoke<{
    agents: any[]
    skills: any[]
    capabilities: any[]
  }>("load_user_manifests")

  const userProgram = Effect.gen(function* () {
    for (const m of userManifests.skills) yield* skillLoader.load(m)
    for (const m of userManifests.agents) yield* agentLoader.load(m)
    for (const m of userManifests.capabilities) yield* capabilityRegistry.install(m)
  })
  await Effect.runPromise(userProgram.pipe(Effect.provide(layer)))

  // 暴露 API 到前端 Store
  ;(window as any).__platform = {
    capabilityRegistry,
    agentLoader,
    skillLoader,
  }
}
```
